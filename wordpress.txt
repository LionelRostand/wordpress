version: '3.7'
services:
  wordpress:
    image: wordpress
	depends_on:
	  - maindb
	volumes:
	  - /mnt/wp:/var/www/html/wp-content
	secrets:
	  - wordpress_db_host
	  - wordpress_db_user
	  - wordpress_db_name
	  - wordpress_db_password
	environment:
	  - wordpress_DB_HOST_FILE=/run/secret/wordpress_db_host
	  - wordpress_DB_USER_FILE=/run/secret/wordpress_db_user
	  - wordpress_DB_NAME_FILE=/run/secret/wordpress_db_name
	  - wordpress_DB_PASSWORD_FILE=/run/secret/wordpress_db_password
	networks:
	  - proxy
	deploy:
	  placement:
	    contraints: [node.role == worker]
	  replicas: 1
	  update_config: 
	    parallelism: 2
		delay: 10s
	  restart_policy:
	    condition: on-failure
	  labels:
	    - "traefik.enable=true"
		- "traefik.docker.networks=proxy
		- "traefik.http.routers.wordpress.rule=Host('www.youtech-consulting.com')"
		- "traefik.http.routers.wordpress.tls=true"
		- "traefik.http.routers.wordpress.tls.certresolver=default"
		- "traefik.http.routers.wordpress.entrypoints=websecure"
		- "traefik.http.services.wordpress.loadbalancer.server.port=80"
secrets:
  wordpress_db_host:
    file: ./wordpress_db_host.txt
  wordpress_db_user:
    file: ./wordpress_db_user.txt
  wordpress_db_name:
    file: ./wordpress_db_name.txt
  wordpress_db_passwor:
    file: ./wordpress_db_password.txt
volumes:
  wp:
    driver: "local"
networks:
  proxy:
    external: true
